# Web Share Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a static HTML chat history viewer with drag-and-drop support and a `parley export` CLI command that produces a self-contained shareable HTML file.

**Architecture:** A single HTML template (`web/share/template.html`) with inlined CSS and JS serves both modes. In standalone mode, users drag-drop room files; in export mode, Go embeds JSON data into the template. The Go export logic lives in `internal/web/export.go`, wired up as a `parley export` subcommand.

**Tech Stack:** Go (embed, html/template), vanilla HTML/CSS/JS (no frameworks), Cobra CLI

---

## File Structure

```
internal/web/
  template.html          — Self-contained HTML with inlined CSS + JS (embedded into Go binary)
  export.go              — Export() function: reads room JSON files, injects into template, writes HTML
  export_test.go         — Unit tests for Export()

cmd/parley/
  main.go                — Add "export" subcommand (modify existing file)
```

```javascript
(function() {
  'use strict';

  // ---- Color assignment ----
  const HUMAN_COLORS = ['#3b82f6', '#2563eb', '#1d4ed8'];
  const AGENT_COLORS = ['#8b5cf6', '#059669', '#d97706', '#dc2626', '#0891b2', '#7c3aed'];
  const GRADIENT_MAP = {
    '#3b82f6': 'linear-gradient(135deg, #3b82f6, #60a5fa)',
    '#2563eb': 'linear-gradient(135deg, #2563eb, #60a5fa)',
    '#1d4ed8': 'linear-gradient(135deg, #1d4ed8, #3b82f6)',
    '#8b5cf6': 'linear-gradient(135deg, #7c3aed, #a78bfa)',
    '#059669': 'linear-gradient(135deg, #059669, #34d399)',
    '#d97706': 'linear-gradient(135deg, #d97706, #fbbf24)',
    '#dc2626': 'linear-gradient(135deg, #dc2626, #f87171)',
    '#0891b2': 'linear-gradient(135deg, #0891b2, #22d3ee)',
    '#7c3aed': 'linear-gradient(135deg, #7c3aed, #a78bfa)',
  };

  function hashName(name) {
    let h = 0;
    for (let i = 0; i < name.length; i++) {
      h = ((h << 5) - h + name.charCodeAt(i)) | 0;
    }
    return Math.abs(h);
  }

  function getColor(name, source) {
    const palette = source === 'human' ? HUMAN_COLORS : AGENT_COLORS;
    return palette[hashName(name) % palette.length];
  }

  function getGradient(color) {
    return GRADIENT_MAP[color] || color;
  }

  // ---- Markdown rendering (minimal) ----
  function renderMarkdown(text) {
    // Escape HTML
    let html = text.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;');
    // Fenced code blocks: ```...```
    html = html.replace(/```(\w*)\n([\s\S]*?)```/g, function(_, lang, code) {
      return '<pre><code>' + code.replace(/\n$/, '') + '</code></pre>';
    });
    // Inline code: `...`
    html = html.replace(/`([^`]+)`/g, '<code>$1</code>');
    // Bold: **...**
    html = html.replace(/\*\*(.+?)\*\*/g, '<strong>$1</strong>');
    // Italic: *...*
    html = html.replace(/(?<!\*)\*(?!\*)(.+?)(?<!\*)\*(?!\*)/g, '<em>$1</em>');
    // Line breaks
    html = html.replace(/\n/g, '<br>');
    // Clean up <br> inside <pre>
    html = html.replace(/<pre><code>([\s\S]*?)<\/code><\/pre>/g, function(_, code) {
      return '<pre><code>' + code.replace(/<br>/g, '\n') + '</code></pre>';
    });
    return html;
  }

  // ---- Mention highlighting ----
  function renderMentions(html, colorMap) {
    return html.replace(/@(\w+)/g, function(match, name) {
      const color = colorMap[name];
      if (color) {
        return '<span class="mention" style="color:' + color + '">@' + name + '</span>';
      }
      return match;
    });
  }

  // ---- Time formatting ----
  function formatTime(ts) {
    if (!ts) return '';
    const d = new Date(ts);
    if (isNaN(d.getTime())) return '';
    return d.toLocaleTimeString([], { hour: 'numeric', minute: '2-digit' });
  }

  function formatDate(ts) {
    if (!ts) return '';
    const d = new Date(ts);
    if (isNaN(d.getTime())) return '';
    return d.toLocaleDateString([], { year: 'numeric', month: 'short', day: 'numeric' });
  }

  // ---- SVG Logo ----
  var LOGO_SVG = '<svg width="28" height="28" viewBox="0 0 28 28" fill="none">'
    + '<rect x="6" y="4" width="18" height="14" rx="4" fill="#4f46e5" opacity="0.6"/>'
    + '<rect x="4" y="8" width="18" height="14" rx="4" fill="#818cf8"/>'
    + '<circle cx="9" cy="15" r="1.5" fill="#1a1a2e"/>'
    + '<circle cx="13" cy="15" r="1.5" fill="#1a1a2e"/>'
    + '<circle cx="17" cy="15" r="1.5" fill="#1a1a2e"/>'
    + '</svg>';

  // ---- Render functions ----
  function renderSidebar(agents) {
    var html = '<div class="logo">'
      + '<div class="logo-row">' + LOGO_SVG
      + '<span class="logo-wordmark">Parley</span></div>'
      + '<div class="logo-tagline">multi-agent group chat</div></div>'
      + '<div class="section-label">Participants</div>';

    if (!agents || agents.length === 0) {
      html += '<div class="sidebar-empty">No data loaded</div>';
      return html;
    }

    agents.forEach(function(a) {
      var color = getColor(a.name, a.source);
      html += '<div class="participant">';
      html += '<div class="participant-header">';
      html += '<span class="participant-name" style="color:' + color + '">' + escapeHtml(a.name) + '</span>';
      if (a.agent_type) {
        html += '<span class="participant-badge" style="background:' + color + '22;color:' + color + '">' + escapeHtml(a.agent_type) + '</span>';
      }
      html += '</div>';
      if (a.role) {
        html += '<div class="participant-role">' + escapeHtml(a.role) + '</div>';
      }
      if (a.directory) {
        html += '<div class="participant-dir">' + escapeHtml(a.directory) + '</div>';
      }
      html += '</div>';
    });

    return html;
  }

  function renderMessages(messages, colorMap) {
    var html = '';
    messages.forEach(function(msg) {
      var color = colorMap[msg.from] || '#94a3b8';
      var gradient = getGradient(color);
      var text = (msg.content || []).map(function(c) { return c.text || ''; }).join('');
      var rendered = renderMentions(renderMarkdown(text), colorMap);

      html += '<div class="message">';
      html += '<div class="avatar" style="background:' + gradient + '">' + escapeHtml(msg.from.charAt(0).toUpperCase()) + '</div>';
      html += '<div class="message-body">';
      html += '<div class="message-header">';
      html += '<span class="message-name">' + escapeHtml(msg.from) + '</span>';
      html += '<span class="message-time">' + formatTime(msg.timestamp) + '</span>';
      html += '</div>';
      html += '<div class="message-content">' + rendered + '</div>';
      html += '</div></div>';
    });
    return html;
  }

  function renderApp(data) {
    var room = data.room || {};
    var agents = data.agents || [];
    var messages = data.messages || [];

    // Build color map
    var colorMap = {};
    agents.forEach(function(a) {
      colorMap[a.name] = getColor(a.name, a.source);
    });
    // Also assign colors for message senders not in agents list
    messages.forEach(function(m) {
      if (!colorMap[m.from]) {
        colorMap[m.from] = getColor(m.from, m.source || 'agent');
      }
    });

    var firstTs = messages.length > 0 ? formatDate(messages[0].timestamp) : '';
    var meta = [firstTs, messages.length + ' messages'].filter(Boolean).join(' \u00b7 ');

    var app = document.getElementById('app');
    app.innerHTML = '<div class="app">'
      + '<div class="sidebar">' + renderSidebar(agents) + '</div>'
      + '<div class="chat">'
      + '<div class="topic-bar"><span class="topic-title">' + escapeHtml(room.topic || 'Untitled') + '</span>'
      + '<span class="topic-meta">' + escapeHtml(meta) + '</span></div>'
      + '<div class="messages">' + renderMessages(messages, colorMap) + '</div>'
      + '<div class="footer">Generated by Parley \u2014 multi-agent group chat</div>'
      + '</div></div>';
  }

  function renderEmpty() {
    var app = document.getElementById('app');
    app.innerHTML = '<div class="app">'
      + '<div class="sidebar">' + renderSidebar(null) + '</div>'
      + '<div class="empty-state">'
      + '<div class="drop-zone">'
      + '<div class="drop-area" id="drop-area">'
      + '<svg width="48" height="48" viewBox="0 0 48 48" fill="none" style="margin:0 auto 16px;display:block">'
      + '<path d="M8 12C8 9.79 9.79 8 12 8H20L24 12H36C38.21 12 40 13.79 40 16V36C40 38.21 38.21 40 36 40H12C9.79 40 8 38.21 8 36V12Z" fill="#e2e8f0" stroke="#94a3b8" stroke-width="1.5"/>'
      + '<path d="M16 24L24 30L32 24" stroke="#94a3b8" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"/>'
      + '<path d="M24 18V30" stroke="#94a3b8" stroke-width="2" stroke-linecap="round"/>'
      + '</svg>'
      + '<div class="drop-title">Drop a Parley room folder here</div>'
      + '<div class="drop-desc">Drag a room folder from <code>~/.parley/rooms/</code><br>'
      + 'containing <code>room.json</code>, <code>messages.json</code>, <code>agents.json</code></div>'
      + '<div class="drop-or">or</div>'
      + '<button class="browse-btn" id="browse-btn">Browse files</button>'
      + '<input type="file" id="file-input" multiple accept=".json" style="display:none">'
      + '</div>'
      + '<div class="file-hint">'
      + '<div class="file-hint-label">Expected files</div>'
      + '<div class="file-hint-list">'
      + '<div class="file-hint-item"><span class="file-hint-dot">\u25A0</span> room.json <span class="file-hint-desc">\u2014 topic &amp; room ID</span></div>'
      + '<div class="file-hint-item"><span class="file-hint-dot">\u25A0</span> messages.json <span class="file-hint-desc">\u2014 chat history</span></div>'
      + '<div class="file-hint-item"><span class="file-hint-dot">\u25A0</span> agents.json <span class="file-hint-desc">\u2014 participants</span></div>'
      + '</div></div>'
      + '</div></div></div>';

    setupDragDrop();
  }

  // ---- Drag and drop ----
  function setupDragDrop() {
    var dropArea = document.getElementById('drop-area');
    var browseBtn = document.getElementById('browse-btn');
    var fileInput = document.getElementById('file-input');

    if (!dropArea) return;

    ['dragenter', 'dragover'].forEach(function(evt) {
      dropArea.addEventListener(evt, function(e) {
        e.preventDefault();
        e.stopPropagation();
        dropArea.classList.add('dragover');
      });
    });
    ['dragleave', 'drop'].forEach(function(evt) {
      dropArea.addEventListener(evt, function(e) {
        e.preventDefault();
        e.stopPropagation();
        dropArea.classList.remove('dragover');
      });
    });

    dropArea.addEventListener('drop', function(e) {
      handleFiles(e.dataTransfer.files);
    });

    browseBtn.addEventListener('click', function() {
      fileInput.click();
    });
    fileInput.addEventListener('change', function() {
      handleFiles(fileInput.files);
    });
  }

  function handleFiles(fileList) {
    var data = { room: null, agents: null, messages: null };
    var remaining = fileList.length;

    if (remaining === 0) return;

    Array.from(fileList).forEach(function(file) {
      var reader = new FileReader();
      reader.onload = function(e) {
        try {
          var json = JSON.parse(e.target.result);
          var name = file.name.toLowerCase();
          if (name === 'room.json') data.room = json;
          else if (name === 'agents.json') data.agents = json;
          else if (name === 'messages.json') data.messages = json;
        } catch(err) {
          console.error('Failed to parse ' + file.name + ':', err);
        }
        remaining--;
        if (remaining === 0 && (data.room || data.messages)) {
          renderApp(data);
        }
      };
      reader.readAsText(file);
    });
  }

  function escapeHtml(str) {
    if (!str) return '';
    return str.replace(/&/g, '&amp;').replace(/</g, '&lt;').replace(/>/g, '&gt;')
              .replace(/"/g, '&quot;').replace(/'/g, '&#39;');
  }

  // ---- Init ----
  function init() {
    var dataEl = document.getElementById('parley-data');
    var content = dataEl ? dataEl.textContent.trim() : '';

    if (content && content !== '{{PARLEY_DATA}}') {
      try {
        var data = JSON.parse(content);
        renderApp(data);
        return;
      } catch(err) {
        console.error('Failed to parse embedded data:', err);
      }
    }
    renderEmpty();
  }

  if (document.readyState === 'loading') {
    document.addEventListener('DOMContentLoaded', init);
  } else {
    init();
  }
})();
```

	}

	html := buf.String()

	// Should contain the full HTML document.
	if !strings.Contains(html, "<!DOCTYPE html>") {
		t.Error("output missing DOCTYPE")
	}

	// Should contain embedded JSON data (not the placeholder).
	if strings.Contains(html, "{{PARLEY_DATA}}") {
		t.Error("output still contains placeholder")
	}

	// Extract and validate the embedded JSON.
	start := strings.Index(html, `<script id="parley-data" type="application/json">`)
	end := strings.Index(html, `</script>`)
	if start == -1 || end == -1 {
		t.Fatal("could not find parley-data script tag")
	}
	jsonStart := start + len(`<script id="parley-data" type="application/json">`)
	jsonStr := html[jsonStart:end]

	var data struct {
		Room     json.RawMessage `json:"room"`
		Agents   json.RawMessage `json:"agents"`
		Messages json.RawMessage `json:"messages"`
	}
	if err := json.Unmarshal([]byte(jsonStr), &data); err != nil {

package web

import (
	"embed"
	"encoding/json"
	"fmt"
	"io"
	"os"
	"path/filepath"
	"strings"
)

//go:embed template.html
var content embed.FS

// templateHTML holds the raw template loaded from the embedded file.
var templateHTML string

func init() {
	data, err := content.ReadFile("template.html")
	if err != nil {
		panic("web: embedded template.html not found: " + err.Error())
	}
	templateHTML = string(data)
}

// placeholder is the marker in the template that gets replaced with real data.
const placeholder = "{{PARLEY_DATA}}"

// Export reads room data from dir (room.json, messages.json, agents.json),
// embeds it into the HTML template, and writes the result to w.
func Export(dir string, w io.Writer) error {
	room, err := readJSONFile(filepath.Join(dir, "room.json"))
	if err != nil {
		return fmt.Errorf("read room.json: %w", err)
	}

	messages, err := readJSONFile(filepath.Join(dir, "messages.json"))
	if err != nil {
		return fmt.Errorf("read messages.json: %w", err)
	}

	// agents.json is optional — some rooms may not have it.
	agents, _ := readJSONFile(filepath.Join(dir, "agents.json"))
	if agents == nil {
		agents = json.RawMessage("[]")
	}

	bundle := struct {
		Room     json.RawMessage `json:"room"`
		Agents   json.RawMessage `json:"agents"`
		Messages json.RawMessage `json:"messages"`
	}{
		Room:     room,
		Agents:   agents,
		Messages: messages,
	}

	bundleJSON, err := json.Marshal(bundle)
	if err != nil {
		return fmt.Errorf("marshal bundle: %w", err)
	}

	output := strings.Replace(templateHTML, placeholder, string(bundleJSON), 1)
	_, err = io.WriteString(w, output)
	return err
}

func readJSONFile(path string) (json.RawMessage, error) {
	data, err := os.ReadFile(path)
	if err != nil {
		return nil, err
	}
	// Validate it's valid JSON.
	if !json.Valid(data) {
		return nil, fmt.Errorf("%s contains invalid JSON", filepath.Base(path))
	}
	return json.RawMessage(data), nil
}

func runExport(cmd *cobra.Command, args []string) error {
	roomID := args[0]
	dir := server.RoomDir(roomID)

	// Check the directory exists.
	if _, err := os.Stat(dir); os.IsNotExist(err) {
		return fmt.Errorf("export: room %q not found at %s", roomID, dir)
	}

	output := exportOutput
	if output == "" {
		output = roomID + ".html"
	}

	f, err := os.Create(output)
	if err != nil {
		return fmt.Errorf("export: create output file: %w", err)
	}
	defer f.Close()

	if err := web.Export(dir, f); err != nil {
		os.Remove(output) // Clean up on error.
		return fmt.Errorf("export: %w", err)
	}

	fmt.Fprintf(os.Stderr, "Exported to %s\n", output)
	return nil
}
